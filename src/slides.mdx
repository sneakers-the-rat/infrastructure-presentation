  )\n\

# no problem, reuse previous metadata to find more!\n\
converter.add_interface(\'recording\',\'open_ephys\',\n\
   file_path = spec.Glob(\'oe_data_{subject_id}\'))\n\
\n\

# No problem, just tell me out to get it!\n\
